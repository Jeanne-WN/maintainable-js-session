Observer pattern - interface

```js function Observer() { } Observer.prototype.update = function() { throw new Error("not implemented"); } ```

Observer pattern - implementation

```js var bbs = Object.create(Observer.prototype); bbs.update = function(data) { console.log(data); } ```

Observer pattern - Subject

```js var Subject = function() { this.observers = []; } Subject.prototype.addObserver = function(observer) { this.observers.push(observer); } Subject.prototype.notify = function(data) { this.observers.forEach(function(observer) { observer.update(data); }); } ```

Observer pattern - usage

```js var weather = new Subject(); weather.addObserver(bbs); weather.notify("Cloudy in the morning"); ```

Observer pattern - usage

```js var alwayson = Object.create(Observer.prototype); alwayson.update = function(data) { alert(data); } ```

Observer pattern - usage

```js var weather = new Subject(); weather.addObserver(bbs); weather.addObserver(alwayson); weather.notify("Cloudy in the morning"); ```
